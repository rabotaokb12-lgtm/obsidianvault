## 1. Python 
### app.py
``` python 
from flask import Flask
import os

app = Flask(__name__)

@app.route("/")
def index():
    return "Hello from Advanced Dockerfile!"

if __name__ == "__main__":
    port = int(os.getenv("PORT", "8080"))
    app.run(host="0.0.0.0", port=port)
```
### dockerfile
``` dockerfile
FROM python:3.9
RUN pip install Flask==3.1.3
ARG BUILD_MODE=dev
LABEL version="0.0.1" \
      maintainer="kotakbas"
RUN useradd -M -r kotakuser
USER kotakuser
VOLUME /home/myapp/data
ONBUILD COPY . /home/myapp/data
STOPSIGNAL SIGQUIT
HEALTHCHECK --interval=30s --timeout=5s --start-period=10s --retries=3 \
  CMD curl -f http://localhost:8080/ || exit 1
SHELL ["/bin/bash", "-c"]
COPY app.py /home/myapp/app.py
EXPOSE 8080
ENTRYPOINT ["python", "/home/myapp/app.py"]
```
